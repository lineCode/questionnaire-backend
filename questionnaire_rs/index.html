[−][src]Crate questionnaire_rs

Backend of the Questionnaire app.

API endpoints available:

/answers

Method: POST

Headers:

Content-type: application/json

Cookies:

auth-cookie: <cookie_value>

Body:

{
  "option_id": 23
}

Response: 200 OK

/answers/{id}

Method: GET

Response:

{
   "id": 47,
   "user_id": 7,
   "created": "2019-11-01T14:30:30",
   "option_id": 23
}

/logout

Method: GET

Headers:

Authorization: token <access_token>

Response: 200 OK

/presentations

Method: POST

Headers:

Content-type: application/json

Cookies:

auth-cookie: <cookie_value>

Body:

{
  "title": "New Presentation"
}

Response: 200 OK

/presentations/{id}

Method: GET

Response:

{
   "id": 47,
   "title": "New Presentation",
   "user_id": 7,
   "created": "2019-11-01T14:30:30"
}

/questions

Method: POST

Headers:

Content type: application/json

Cookies:

auth-cookie: <cookie_value>

Body:

{
   "title": "New Question",
   "presentation_id": 1,
}

Response: 200 OK

/questions/{id}

Method: GET

Response:

{
   "id": 23,
   "title": "New Question",
   "created": "2019-11-01T14:30:30",
   "presentation_id": 3,
   "user_id": 7,
}
